    function setValue(id, val, triggerChange = false) {
        if (!val) {
            return;
        }
        id = '#' + id;
        $(id).val(val);
        if (triggerChange) {
            $(id).change();
        }
        $(id).attr('disabled', true).attr('readonly', true);
    }

    function selectValue(elm, value) {
        if (!value) {
            return;
        }
        elm.filter(function() {
            return $(this).text() && value && $(this).text().toString().trim().toUpperCase() == value.toString().trim().toUpperCase();
        }).prop('selected', true);

        if (elm.parent().val()) {
            elm.parent().attr('disabled', true).attr('readonly', true);
        }
    };
